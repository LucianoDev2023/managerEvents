// hooks/useMyGuestEvents.ts
import { useEffect, useMemo, useState } from 'react';
import { getAuth } from 'firebase/auth';
import { doc, getDoc } from 'firebase/firestore';
import { useEvents } from '@/context/EventsContext';
import { db } from '@/config/firebase';
import type { Event } from '@/types';
import type { GuestMode } from '@/types/guestParticipation';
import { getGuestParticipationsByUserId } from '@/hooks/guestService';

type GuestEvent = { event: Event; mode: GuestMode };

function mapEvent(docId: string, data: any): Event {
  const { id: _ignoredId, ...rest } = data ?? {};
  return { id: docId, ...(rest as Omit<Event, 'id'>) } as Event;
}

async function fetchEventById(eventId: string): Promise<Event | null> {
  const ref = doc(db, 'events', eventId);
  const snap = await getDoc(ref);
  return snap.exists() ? mapEvent(snap.id, snap.data()) : null;
}

export function useMyGuestEvents() {
  const { state } = useEvents();
  const userUid = getAuth().currentUser?.uid ?? null;

  const [loading, setLoading] = useState(true);
  const [guestEvents, setGuestEvents] = useState<GuestEvent[]>([]);

  const eventsById = useMemo(() => {
    const map = new Map<string, Event>();
    for (const e of state.events) map.set(e.id, e);
    return map;
  }, [state.events]);

  useEffect(() => {
    let cancelled = false;

    const run = async () => {
      if (!userUid) {
        setGuestEvents([]);
        setLoading(false);
        return;
      }

      setLoading(true);
      try {
        const parts = await getGuestParticipationsByUserId(userUid);

        // Busca do Firestore só para os que não estão no cache
        const merged = await Promise.all(
          parts.map(async (p) => {
            const cached = eventsById.get(p.eventId);
            const ev = cached ?? (await fetchEventById(p.eventId));
            if (!ev) return null;
            return { event: ev, mode: p.mode } as GuestEvent;
          })
        );

        const clean = merged.filter(Boolean) as GuestEvent[];
        if (!cancelled) setGuestEvents(clean);
      } catch (e) {
        console.error('useMyGuestEvents:', e);
        if (!cancelled) setGuestEvents([]);
      } finally {
        if (!cancelled) setLoading(false);
      }
    };

    run();
    return () => {
      cancelled = true;
    };
  }, [userUid, eventsById]);

  return { guestEvents, loading };
}
